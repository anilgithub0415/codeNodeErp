import { Repository } from 'typeorm';
import { AppDataSource } from '../../data-source'; // Correct relative path
import { RefreshToken } from '../entity/RefreshToken'; // Import the new entity

export class RefreshTokenRepositoryTypeorm {
    private refreshTokenRepository: Repository<RefreshToken>;

    constructor() {
        if (!AppDataSource.isInitialized) {
            console.warn('AppDataSource not initialized in RefreshTokenRepository. Attempting to initialize...');
            AppDataSource.initialize().catch(error => {
                console.error("Failed to initialize AppDataSource in RefreshTokenRepository constructor:", error);
                process.exit(1);
            });
        }
        this.refreshTokenRepository = AppDataSource.getRepository(RefreshToken);
    }

    /**
     * Saves a new refresh token or updates an existing one.
     * TypeORM handles insert/update based on primary key presence.
     */
    async save(refreshToken: RefreshToken): Promise<RefreshToken> {
        try {
            return await this.refreshTokenRepository.save(refreshToken);
        } catch (error: any) {
            console.error('Error saving refresh token:', error);
            throw error;
        }
    }

    /**
     * Finds a refresh token by its unique token string.
     */
    async findByToken(token: string): Promise<RefreshToken | undefined> {
        try {
            var refreshTokenOBJ= await this.refreshTokenRepository.findOne({ where: { token: token } });
            if (refreshTokenOBJ){
                return refreshTokenOBJ
            }
            else return undefined;
        } catch (error: any) {
            console.error('Error finding refresh token by token string:', error);
            throw error;
        }
    }

    /**
     * Deletes a refresh token by its unique token string.
     */
    async deleteByToken(token: string): Promise<void> {
        try {
            await this.refreshTokenRepository.delete({ token: token });
        } catch (error: any) {
            console.error('Error deleting refresh token by token string:', error);
            throw error;
        }
    }

    /**
     * Deletes all expired refresh tokens for a given user.
     * Useful for cleanup.
     */
    async deleteExpiredTokensForUser(userId: number): Promise<void> {
        try {
            await this.refreshTokenRepository.createQueryBuilder()
                .delete()
                .where("userId = :userId", { userId })
                .andWhere("expiresAt <= :now", { now: new Date() })
                .execute();
        } catch (error: any) {
            console.error('Error deleting expired tokens for user:', error);
            throw error;
        }
    }

    /**
     * Cleans up all expired refresh tokens from the database.
     * Call this periodically (e.g., via a cron job).
     */
    async deleteAllExpiredTokens(): Promise<void> {
        try {
            await this.refreshTokenRepository.createQueryBuilder()
                .delete()
                .where("expiresAt <= :now", { now: new Date() })
                .execute();
            console.log('Expired refresh tokens cleaned up from DB.');
        } catch (error: any) {
            console.error('Error deleting all expired tokens:', error);
            // Don't re-throw for a cleanup job, just log
        }
    }


    // --- NEW METHOD: Delete an existing token for a specific user and device ---
    async deleteExistingTokenForDevice(userId: number, deviceInfo: string): Promise<void> {
        try {
            await this.refreshTokenRepository.delete({
                userId: userId,
                deviceInfo: deviceInfo,
            });
        } catch (error: any) {
            console.error(`Error deleting existing token for user ${userId} on device ${deviceInfo}:`, error);
            throw error;
        }
    }
    // --- END NEW METHOD ---
}

export default RefreshTokenRepositoryTypeorm;

 const RefreshAccessToken = async (refreshToken: string, userId: number): Promise<{ accessToken: string; newRefreshToken?: string; role: string; userId: number }> => {

const settingsService = getSettingsService(); // Access SettingsService via getter

    try {
        // 1. Find the refresh token in the database
        const storedToken = await refreshTokenRepo.findByToken(refreshToken);

        if (!storedToken || storedToken.userId !== userId) {
            throw new Error('Invalid or unauthorized refresh token.');
        }

        // 2. Check if the refresh token is expired
        if (new Date() > storedToken.expiresAt) {
            // Optional: Delete the expired token from DB
            await refreshTokenRepo.deleteByToken(refreshToken);
            throw new Error('Refresh token expired.');
        }

        // 3. (Optional but recommended): Invalidate the old refresh token
        // This enhances security by preventing replay attacks
        await refreshTokenRepo.deleteByToken(refreshToken);

        // 4. Issue a new Access Token
        const user = await userRepo.getById(userId);
        if (!user) {
            throw new Error('User associated with refresh token not found.');
        }

        const payload: JwtPayload = {
            UserName: user.UserName,
            Role: user.Role,
            userId: user.id
        };

        const currentAccessTokenLifetime = settingsService.getSettings().accessTokenLifetimeSeconds;
        const accessToken = jwt.sign(payload, ACCESS_TOKEN_SECRET, { expiresIn: currentAccessTokenLifetime });

        // 5. (Optional): Issue a new Refresh Token (rotating refresh tokens for enhanced security)
        const currentRefreshTokenLifetime = settingsService.getSettings().refreshTokenLifetimeSeconds;
        const newRefreshTokenString = uuidv4();
        const newExpiresAt = new Date(Date.now() + currentRefreshTokenLifetime * 1000);

        const newRefreshToken = new RefreshToken();
        newRefreshToken.token = newRefreshTokenString;
        newRefreshToken.userId = user.id;
        newRefreshToken.expiresAt = newExpiresAt;
        newRefreshToken.deviceInfo = storedToken.deviceInfo; // Inherit device info
        await refreshTokenRepo.save(newRefreshToken);

        return {
            accessToken: accessToken,
            newRefreshToken: newRefreshTokenString,
            role: user.Role,
            userId: user.id
        };

    } catch (error: any) {
        console.error('Error refreshing access token:', error.message || error);
        throw error;
    }

class SettingsService {
    // Private property to hold the cached settings
    private settings: AppSettings = {
        accessTokenLifetimeSeconds: 3600, // Default value (1 hour)
        refreshTokenLifetimeSeconds: 86400 // Default value (24 hours)
    };
    private isInitialized = false;
    private settingsRepository!: SettingsRepositoryTypeorm;

    constructor() {
        this.settingsRepository = new SettingsRepositoryTypeorm();

        // Subscribe to events. Ensure `this.loadSettings` has the correct context.
        eventemitter.on(globalEvents.accessTokenExpiryChanged, () => {
            console.log('accessTokenExpiryChanged Event is occurred .........................');
            // Call loadSettings and await it, or handle it as a background task
            // Ensure `this` context is preserved, e.g., using arrow function or .bind(this)
            this.loadSettings().catch(error => {
                console.error('Error reloading settings on event:', error);
            });
        });
    }

   
    //This loadSettings method is called from 
    //1. LoginServiceTypeORm and 2. constructor of this service on emit of 'globalEvents.accessTokenExpiryChanged' event

    // This reads settings from DB
    public async loadSettings(): Promise<void> { console.log('Running loadSettings.....................................................');
    
        // // Ensure the repository instance is ready before use
        // if (!settingsRepository) {
        //     console.error('SettingsRepository not initialized. Cannot load settings.');
        //     throw new Error('SettingsRepository not initialized.'); // Or handle more gracefully
        // }

        console.log('Loading application settings...');
        try {
            // --- FIX: Directly query the database using the repository instance ---
            // Fetch the single settings record using its primary key 'global_settings'
            let dbSettings = await this.settingsRepository.getById('global_settings');

            if (!dbSettings) {
                // If no record exists, create and save a default one
                console.log('No global settings found in DB. Creating default settings record...');
                dbSettings = {
                    settingKey: 'global_settings',
                    accessTokenLifetime: this.settings.accessTokenLifetimeSeconds, // Use current default
                    refreshTokenLifetime: this.settings.refreshTokenLifetimeSeconds // Use current default
                };
                dbSettings = await this.settingsRepository.save(dbSettings); // Save the new record
                console.log('Default settings record created in DB.');
            }

            // Update the cached settings with values from the database
            this.settings.accessTokenLifetimeSeconds = dbSettings.accessTokenLifetime;
            this.settings.refreshTokenLifetimeSeconds = dbSettings.refreshTokenLifetime; // NEW: Update refresh token expiry

            this.isInitialized = true;
            console.log(`Settings loaded: Access Token Lifetime = ${this.settings.accessTokenLifetimeSeconds} seconds`);
            console.log(`Settings loaded: Refresh Token Lifetime = ${this.settings.refreshTokenLifetimeSeconds} seconds`);

        } catch (error) {
            console.error('Failed to load application settings:', error);
            throw error; // Re-throw to prevent app startup if critical for initialization
        }
    }

    /**
     * Returns the current cached application settings.
     * Ensure loadSettings() has been called prior to accessing for updated values.
     */
    public getSettings(): AppSettings {
        if (!this.isInitialized) {
            console.warn('SettingsService not fully initialized. Returning cached default settings.');
        }
        return this.settings;
    }

    /**
     * Triggers a reload of settings from the database. Useful for admin endpoints.
     */
    public async refreshSettings(): Promise<void> {
        await this.loadSettings();
    }

    /**
     * Saves updated settings to the database.
     * Use this method when an admin explicitly updates settings via an API.
     */
    public async saveSettings(newAccessTokenLifetime: number, newRefreshTokenLifetime: number): Promise<Settings> {
        // if (!settingsRepositoryInstance) {
        //     console.error('SettingsRepository not initialized. Cannot save settings.');
        //     throw new Error('SettingsRepository not initialized.');
        // }

        // Fetch the existing record to update it, or create if it doesn't exist
        let dbSettings = await this.settingsRepository.getById('global_settings');
        if (!dbSettings) {
            dbSettings = new Settings(); // Create a new instance of the entity
            dbSettings.settingKey = 'global_settings';
        }

        dbSettings.accessTokenLifetime = newAccessTokenLifetime;
        dbSettings.refreshTokenLifetime = newRefreshTokenLifetime;

        const savedSettings = await this.settingsRepository.save(dbSettings);
        
        // After saving to DB, immediately update the in-memory cache
        this.settings.accessTokenLifetimeSeconds = savedSettings.accessTokenLifetime;
        this.settings.refreshTokenLifetimeSeconds = savedSettings.refreshTokenLifetime;
        

        return savedSettings;
    }
}
